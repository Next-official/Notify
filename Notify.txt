<script>
    // ---- Persistenza con IndexedDB (sostituisce localStorage) ----
    const IDB_NAME = 'appleJournalDB';
    const IDB_STORE = 'entries';
    const IDB_SETTINGS_STORE = 'settings';
    let idbConn = null;
    let entries = [];
    let editingId = null;

    function openIDB() {
        return new Promise((resolve, reject) => {
            const req = indexedDB.open(IDB_NAME, 2);
            req.onupgradeneeded = (e) => {
                const database = e.target.result;
                if (!database.objectStoreNames.contains(IDB_STORE)) {
                    database.createObjectStore(IDB_STORE, { keyPath: 'id' });
                }
                if (!database.objectStoreNames.contains(IDB_SETTINGS_STORE)) {
                    database.createObjectStore(IDB_SETTINGS_STORE, { keyPath: 'key' });
                }
            };
            req.onsuccess = (e) => resolve(e.target.result);
            req.onerror = (e) => reject(e.target.error);
        });
    }

    function getSetting(key) {
        return new Promise((resolve, reject) => {
            const tx = idbConn.transaction(IDB_SETTINGS_STORE, 'readonly');
            const req = tx.objectStore(IDB_SETTINGS_STORE).get(key);
            req.onsuccess = () => resolve(req.result ? req.result.value : null);
            req.onerror = () => reject(req.error);
        });
    }

    function setSetting(key, value) {
        return new Promise((resolve, reject) => {
            const tx = idbConn.transaction(IDB_SETTINGS_STORE, 'readwrite');
            tx.objectStore(IDB_SETTINGS_STORE).put({ key, value });
            tx.oncomplete = () => resolve();
            tx.onerror = () => reject(tx.error);
        });
    }

    function idbGetAll() {
        return new Promise((resolve, reject) => {
            const tx = idbConn.transaction(IDB_STORE, 'readonly');
            const store = tx.objectStore(IDB_STORE);
            const req = store.getAll();
            req.onsuccess = () => resolve(req.result || []);
            req.onerror = () => reject(req.error);
        });
    }

    function idbPut(entry) {
        return new Promise((resolve, reject) => {
            const tx = idbConn.transaction(IDB_STORE, 'readwrite');
            const store = tx.objectStore(IDB_STORE);
            store.put(entry);
            tx.oncomplete = () => resolve();
            tx.onerror = () => reject(tx.error);
        });
    }

    async function initApp() {
        try {
            idbConn = await openIDB();
            entries = await idbGetAll();
            entries.sort((a, b) => b.id - a.id); // più recenti in cima

            const savedTheme = await getSetting('theme');
            applyTheme(savedTheme === 'light' ? 'light' : 'dark');
        } catch (err) {
            console.error('Errore IndexedDB:', err);
            entries = [];
        }
        renderFeed();
    }

    function exportData() {
        const blob = new Blob([JSON.stringify(entries, null, 2)], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'diario-backup.json';
        document.body.appendChild(a);
        a.click();
        a.remove();
        URL.revokeObjectURL(url);

    }

    function importData(event) {
        const file = event.target.files[0];
        if (!file) return;
        const reader = new FileReader();
        reader.onload = async (e) => {
            try {
                const imported = JSON.parse(e.target.result);
                if (!Array.isArray(imported)) throw new Error('Formato non valido');
                for (const item of imported) {
                    if (item && item.id) {
                        await idbPut(item);
                    }
                }
                entries = await idbGetAll();
                entries.sort((a, b) => b.id - a.id);
                renderFeed();
                closeSidebar();
            } catch (err) {
                console.error('Errore importazione:', err);
                alert('Il file selezionato non è un backup valido.');
            }
        };
        reader.readAsText(file);
        event.target.value = '';
    }

    async function saveJournalEntry() {
        const title = document.getElementById('diary-title').value.trim();
        const body = document.getElementById('diary-body').value.trim();
        if (!title && !body) { closeJournalEditor(); return; }

        // Modalità modifica: aggiorna una voce esistente invece di crearne una nuova
        if (editingId) {
            const idx = entries.findIndex(e => e.id === editingId);
            if (idx !== -1) {
                entries[idx].title = title || "Nessun titolo";
                entries[idx].body = body || "";
                entries[idx].wordCount = body.split(/\s+/).filter(Boolean).length + title.split(/\s+/).filter(Boolean).length;
                try { await idbPut(entries[idx]); } catch (err) { console.error('Errore aggiornamento IndexedDB:', err); }
            }
            editingId = null;
            closeJournalEditor();
            renderFeed();
            return;
        }

        const now = new Date();
        const newEntry = {
            id: Date.now(),
            title: title || "Nessun titolo",
            body: body || "",
            wordCount: body.split(/\s+/).filter(Boolean).length + title.split(/\s+/).filter(Boolean).length,
            dateStr: now.toLocaleDateString('it-IT', { weekday: 'long', day: 'numeric', month: 'short' }),
            dateISO: now.toISOString().slice(0, 10), // es. "2026-07-18", usata per i calcoli statistici
            reaction: null
        };

        entries.unshift(newEntry);
        try {
            await idbPut(newEntry);
        } catch (err) {
            console.error('Errore salvataggio IndexedDB:', err);
        }

        closeJournalEditor();
        renderFeed();
    }

    // Restituisce il numero della settimana ISO (anno-settimana) di una data "YYYY-MM-DD"
    function weekKeyOf(dateISO) {
        const d = new Date(dateISO + 'T00:00:00');
        const oneJan = new Date(d.getFullYear(), 0, 1);
        const week = Math.ceil((((d - oneJan) / 86400000) + oneJan.getDay() + 1) / 7);
        return { year: d.getFullYear(), week };
    }

    function computeStats() {
        const totalEntries = entries.length;
        const totalWords = entries.reduce((acc, curr) => acc + curr.wordCount, 0);
        const uniqueDays = [...new Set(entries.map(e => e.dateISO))].sort();
        const journaledDays = uniqueDays.length;

        // Streak giornaliera più lunga (giorni consecutivi con almeno una voce)
        let longestDaily = 0, run = 0, prevDate = null;
        uniqueDays.forEach(dStr => {
            const d = new Date(dStr + 'T00:00:00');
            run = (prevDate && Math.round((d - prevDate) / 86400000) === 1) ? run + 1 : 1;
            longestDaily = Math.max(longestDaily, run);
            prevDate = d;
        });

        // Streak corrente: giorni consecutivi terminanti oggi (o ieri, se oggi non ancora scritto)
        const daySet = new Set(uniqueDays);
        const today = new Date(); today.setHours(0, 0, 0, 0);
        let cursor = new Date(today);
        if (!daySet.has(cursor.toISOString().slice(0, 10))) cursor.setDate(cursor.getDate() - 1);
        let currentStreak = 0;
        while (daySet.has(cursor.toISOString().slice(0, 10))) {
            currentStreak++;
            cursor.setDate(cursor.getDate() - 1);
        }

        // Streak settimanale più lunga (settimane consecutive con almeno una voce)
        const weekKeys = [...new Set(uniqueDays.map(weekKeyOf).map(w => w.year + '-W' + w.week))].sort();
        let longestWeekly = 0, wrun = 0, prevW = null;
        weekKeys.forEach(wk => {
            const [y, w] = wk.split('-W').map(Number);
            wrun = (prevW && ((y === prevW.y && w === prevW.w + 1) || (y === prevW.y + 1 && w === 1 && prevW.w >= 52))) ? wrun + 1 : 1;
            longestWeekly = Math.max(longestWeekly, wrun);
            prevW = { y, w };
        });

        // Parole scritte questo mese / quest'anno
        const now = new Date();
        let wordsMonth = 0, wordsYear = 0;
        entries.forEach(e => {
            const d = new Date(e.dateISO + 'T00:00:00');
            if (d.getFullYear() === now.getFullYear()) {
                wordsYear += e.wordCount;
                if (d.getMonth() === now.getMonth()) wordsMonth += e.wordCount;
            }
        });

        // Istogramma: numero di voci negli ultimi 6 giorni (oggi incluso)
        const last6 = [];
        for (let i = 5; i >= 0; i--) {
            const d = new Date(today); d.setDate(d.getDate() - i);
            const iso = d.toISOString().slice(0, 10);
            last6.push(entries.filter(e => e.dateISO === iso).length);
        }

        return { totalEntries, totalWords, journaledDays, currentStreak, longestDaily, longestWeekly, wordsMonth, wordsYear, last6 };
    }

    function formatK(n) {
        return n >= 1000 ? (n / 1000).toFixed(n >= 10000 ? 0 : 1).replace('.0', '') + 'K' : String(n);
    }

    function renderBarChart(last6) {
        const container = document.getElementById('entries-bar-chart');
        container.innerHTML = '';
        const max = Math.max(...last6, 1);
        last6.forEach((count, i) => {
            const bar = document.createElement('div');
            const isLast = i === last6.length - 1;
            bar.className = 'chart-bar' + (isLast ? ' active' : '');
            const heightPct = count === 0 ? 8 : Math.max(15, Math.round((count / max) * 100));
            bar.style.height = heightPct + '%';
            container.appendChild(bar);
        });
    }

    function recalculateInsights() {
        if (entries.length === 0) {
            // Nessuna voce salvata: mostro i valori dimostrativi originali
            document.getElementById('stat-total-entries').innerText = 43;
            document.getElementById('stat-total-words').innerText = (19007).toLocaleString('en-US');
            document.getElementById('stat-journaled-days').innerText = 19;
            document.getElementById('stat-current-streak').innerText = 6;
            document.getElementById('stat-longest-daily').innerText = 6;
            document.getElementById('stat-longest-weekly').innerText = 4;
            document.getElementById('stat-words-month').innerText = '15K';
            document.getElementById('stat-words-year').innerText = '19K';
            return;
        }

        const s = computeStats();
        document.getElementById('stat-total-entries').innerText = s.totalEntries;
        document.getElementById('stat-total-words').innerText = s.totalWords.toLocaleString('en-US');
        document.getElementById('stat-journaled-days').innerText = s.journaledDays;
        document.getElementById('stat-current-streak').innerText = s.currentStreak;
        document.getElementById('stat-longest-daily').innerText = s.longestDaily;
        document.getElementById('stat-longest-weekly').innerText = s.longestWeekly;
        document.getElementById('stat-words-month').innerText = formatK(s.wordsMonth);
        document.getElementById('stat-words-year').innerText = formatK(s.wordsYear);
        renderBarChart(s.last6);
    }
